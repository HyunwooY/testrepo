drop index ix_auth_username;
drop table myauthorities;
drop table myusers;
create table myusers
(
	userid varchar2(15) not null primary key,
	username varchar2(50) not null,
	password varchar2(100) not null,
	regdate date,
	enabled chr(1) default '1'
);
create table myauthorities
(
	userid varchar2(15) not null,
	authority varchar2(50) not null,
	constraint fk_mauthorities_users foreign key(userid) references myusers(userid)
);
create unique index ix_auth_username on myauthorities (userid,authority);
